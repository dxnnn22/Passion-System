    <script>
        /* JavaScript Code for Interactivity */
        function submitOrder() {
            // Get values from the form inputs
            const name = document.getElementById('customerName').value;
            const type = document.getElementById('printType').value;
            const quantity = document.getElementById('quantity').value;
            const fileInput = document.getElementById('designFile');
            
            // Simple validation
            if (name === "" || quantity < 1) {
                alert("الرجاء إدخال الاسم والكمية بشكل صحيح.");
                return;
            }

            // Get file name if one was selected (simple example)
            const fileName = fileInput.files.length > 0 ? fileInput.files[0].name : "لم يتم تحميل ملف";

            // Create summary text
            const summaryText = `
                الاسم: ${name}<br>
                النوع: ${type}<br>
                الكمية: ${quantity}<br>
                الملف: ${fileName}
            `;

            // Display the summary section
            document.getElementById('summaryDetails').innerHTML = summaryText;
            document.getElementById('orderSummary').style.display = 'block';

            // In a real system, you would send this data to a server using fetch() or AJAX here.
            alert("تم استلام طلبك بنجاح! يمكنك رؤية الملخص أدناه.");
        }
